class program {

    void regresar(){
        iterate(2)
            turnleft();
         move();
         turnleft();
         move();
    }
    void tomar(){
    	while(nextToABeeper){
        	pickbeeper();
            move();
            }
    }
    program(){
    	while(nextToABeeper){
    		tomar();
    		iterate(3){
    			regresar();
        		tomar();  
        	}
            regresar();
        }
        
        turnoff();
    }
}
